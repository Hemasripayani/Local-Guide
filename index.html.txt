        async function searchGuides() {
            const city = document.getElementById('cityInput').value.trim();
            const date = document.getElementById('dateInput').value;
            const time = document.getElementById('timeInput').value;
            const budget = document.getElementById('budgetSlider').value;
            
            // Validation
            if (!city) {
                showAlert('Please enter a destination city', 'error');
                return;
            }
            
            if (!date) {
                showAlert('Please select a travel date', 'error');
                return;
            }
            
            // Save search data
            searchData = { city, date, time, budget };
            
            showLoading('Finding the best guides for you...', 'Searching database');
            
            try {
                let guides;
                
                if (USE_MOCK_DATA) {
                    // Mock API call with delay
                    await new Promise(resolve => setTimeout(resolve, 1200));
                    guides = mockGuides.filter(g => 
                        g.city.toLowerCase().includes(city.toLowerCase()) && 
                        g.price_per_day <= parseInt(budget)
                    );
                    
                    console.log('‚úì Mock Search:', { city, budget, results: guides.length });
                    
                } else {
                    // Real API call
                    const response = await fetch(API_BASE_URL + 'search_guides.php', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            city: city,
                            budget: parseInt(budget),
                            date: date
                        })
                    });
                    
                    const result = await response.json();
                    
                    console.log('API Response:', result);
                    
                    if (!result.success) {
                        throw new Error(result.message || 'Search failed');
                    }
                    
                    guides = result.data.guides || [];
                }
                
                allGuides = guides;
                filteredGuides = guides;
                
                hideLoading();
                
                if (guides.length > 0) {
                    displayGuides(guides);
                    goToFrame(2);
                } else {
                    displayGuides([]);
                    goToFrame(2);
                    showAlert('No guides found. Try adjusting your budget or search city.', 'info');
                }
                
            } catch (error) {
                console.error('‚ùå Search Error:', error);
                hideLoading();
                showAlert('Failed to search guides. Please check your connection and try again.', 'error');
            }
        }

        function filterGuides(filterType, event) {
            // Update active chip
            document.querySelectorAll('.chip').forEach(chip => {
                chip.classList.remove('active');
            });
            event.target.classList.add('active');
            
            let sorted = [...allGuides];
            
            switch(filterType) {
                case 'top-rated':
                    sorted.sort((a, b) => b.rating - a.rating);
                    break;
                case 'low-price':
                    sorted.sort((a, b) => a.price_per_day - b.price_per_day);
                    break;
                case 'verified':
                    sorted = sorted.filter(g => g.is_verified);
                    break;
                default:
                    // 'all' - keep original order
                    break;
            }
            
            filteredGuides = sorted;
            displayGuides(sorted);
        }
